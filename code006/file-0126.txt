<?php

declare(strict_types=1);

namespace App\Base;

use App\Models\AbstractModel;
use Illuminate\Database\Eloquent\Builder;
use Illuminate\Database\Eloquent\Model;
use Illuminate\Pagination\LengthAwarePaginator;
use Illuminate\Pagination\Paginator;

class ReadPageQueryBuilder
{
    /** @phpstan-ignore-next-line */
    public function __construct(public Builder $Builder, public PageParameter $PageParameter) {}

    public function addFilter(string $Key): self
    {
        $value = $this->PageParameter->getFilterValue($Key);
        if (isset($value)) {
            $this->Builder->where($Key, $value);
        }

        return $this;
    }

    /**
     * @param  string[]  $Keyzz
     * @return $this
     */
    public function addFilterzz(array $Keyzz): self
    {
        foreach ($Keyzz as $Key) {
            $this->addFilter($Key);
        }

        return $this;
    }

    public function addSort(): self
    {
        foreach ($this->PageParameter->Sortzz as $item) {
            $SortField = $item['Column'];
            if ($SortField) {
                $this->Builder->orderBy($SortField, $item['Order'] ?? PageParameter::SortAsc);
            }
        }

        if (count($this->PageParameter->Sortzz) === 0) {
            $this->Builder->orderBy(AbstractModel::PrimaryKey, PageParameter::SortDesc);
        }

        return $this;
    }

    /**
     * @return LengthAwarePaginator<int, Model>
     */
    public function paginate(): LengthAwarePaginator
    {
        return $this->Builder->paginate(
            $this->PageParameter->PerPage,
            ['*'],
            'Page',
            $this->PageParameter->Page,
        );
    }

    /**
     * @return Paginator<int, Model>
     */
    public function paginateSimple(): Paginator
    {
        return $this->Builder->simplePaginate(
            $this->PageParameter->PerPage,
            ['*'],
            'Page',
            $this->PageParameter->Page,
        );
    }
}
