        return true;
    }

    /**
     * @return array<string, string[]>
     */
    public function rules(): array
    {
        return static::getRulezz();
    }

    /**
     * @return array<string, string[]>
     */
    public static function getRulezzForItemzz(): array
    {
        $rulezz = [
            'itemzz' => ['array', 'max:111', 'required'],
        ];
        foreach (static::getRulezz() as $key => $item) {
            $rulezz['itemzz.*.'.$key] = $item;
        }

        return $rulezz;
    }

    /**
     * @return array<string, string[]>
     */
    abstract public static function getRulezz(): array;
}
